 12 	    Wrap a Node of type Feed or Entry
 13 	    @class 
 14 	    @param node XMLNode of an entry or feed
 15 	 */
 16 	CMIS.Node = function(node) {
 17 		var that = this;
 18 		
 19 		/** The XML node wrapped. */
 20 		this.xmlNode = node;
 21 		
 22 		/** The title of the feed or Entry. */
 23 		this.title = this.xmlNode.children("title").eq(0).text();
 24 		
 25 		/** The properties of the feed or Entry. */
 26 		this.properties = {};
 27 		this._entry_cache_by_id = null;
 28 		this._entries_cache = null;
 29 
 30 		_getPropertiesNode(this.xmlNode).children().each(function(i, node) {
 31 			that.properties[$(node).attr("propertyDefinitionId")] = $(node).find("[nodeName=cmis:value]").text();
 32 		});
 33 		
 34 		if (this.getProperty("cmis:objectId")) {
 35 			this.id = this.getProperty("cmis:objectId");
 36 		} else {
 37 			this.id = this.xmlNode.children("id").text();
 38 		}
 39 	}

 41 	/** Get the links for the current Node */
 42 	CMIS.Node.prototype.getLinks = function(rel, type) {
 43 		
 44 		var selector = "link", res = [];
 45 		if (rel !== undefined) {
 46 			selector += "[rel=" + rel + "]";
 47 		}
 48 		
 49 		if (type !== undefined) {
 50 			selector += "[type=" + type + "]";
 51 		}
 52 		this.xmlNode.children(selector).each(function(i, n) {
 53 			res.push($(n).attr("href"));
 54 		});
 55 		return res;
 56 	}

 95 	/**
 96 	    Return all the entries in a feed
 97 	 */
 98 	CMIS.Node.prototype.getEntries = function() {
 99 		var that = this;
100 		if (!this.isFeed()) {
101 			return [];
102 		}
103 		
104 		if (this._entries_cache === null) {
105 			this._entries_cache = [];
106 			this.xmlNode.children("entry").each(function(i, node) {
107 				that._entries_cache.push(new CMIS.Node($(node)));
108 			});
109 			this._entries_cache = $(this._entries_cache);
110 		}
111 		
112 		return this._entries_cache;
113 	}

124 	/**
125 	    Return the entry of a given ID from a feed
126 	 */
127 	CMIS.Node.prototype.getEntry = function(id) {
128 		if (!this.isFeed()) {
129 			return null;
130 		}
131 		if (this._entry_cache_by_id === null) {
132 			var that = this;
133 			this._entry_cache_by_id = {};
134 			this.getEntries().each(function(i, entry) {
135 				that._entry_cache_by_id[entry.id] = entry;
136 			});
137 		}
138 		return id in this._entry_cache_by_id ? this._entry_cache_by_id[id] : null;
139 	}

164 	/**
165 	    Parse a feed or and entry
166 	 */
167 	CMIS.parse = function(/* XMLDocument */ xml) {
168 		if ($(xml).children("feed").length > 0) {
169 			return new CMIS.Node($(xml).children("feed").eq(0));
170 		} else if ($(xml).children("service").length > 0) {
171 		    var res = [];
172 		    $(xml).children("service").children("workspace").each(function(i, workspace) {
173 		        res.push(new CMIS.Workspace($(workspace)));
174 		    });
175 		    
176 		    return res;
177 		}
178 		return new CMIS.Node($(xml).children("entry").eq(0));
179 	}

188 	var _getPropertiesNode = function(node) {
189 		// this doesn't work due to the namespace, so we use another way arround
190 		// var props = node.children("cmisra\\:object > cmis\\:properties");
191 		var res = null;
192 		propsNodes = $(node).find("[nodeName=cmis:properties]").each(function(i, propsNode) {
193 			if ($(propsNode).parent().parent()[0] === $(node)[0]) {
194 				res = $(propsNode);
195 				return false;
196 			}
197 		});
198 		return res || $();
